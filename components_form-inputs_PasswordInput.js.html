// TOP LEVEL IMPORTS
import React from 'react';
import PropTypes from 'prop-types';
import { View, StyleSheet, Keyboard, Text } from 'react-native';
// LIB
import { stylesConfig, colorConfig, appConfig, fontConfig } from '../../lib/config';
//
import { TextField } from 'react-native-material-textfield';


// DOCS
// =========================
/**
 *
 * @namespace PasswordInput
 * @description reusable password input
 * @prop label {string}
 * @prop value {string}
 * @prop placeholder {string}
 * @prop onChange {string}
 * @prop containerStyle {object}
 * @prop inputStyle {object}
 * 
 */


// EXPORTED COMPONENT
// =========================
class PasswordInput extends React.PureComponent {
  state = {
    secure: true,
  };
  onToggleSecure = () => {
    this.setState({ secure: !this.state.secure });
  };
  render() {
    const { label, value, placeholder, onChange, containerStyle, inputStyle } = this.props;
    return (
      <View style={{ position: 'relative' }}>
        <TextField
          secureTextEntry={this.state.secure}
          containerStyle={containerStyle}
          defaultValue={value}
          label={label}
          tintColor={colorConfig.primary}
          textColor={colorConfig.primary}
          style={inputStyle}
          value={value}
          onChangeText={onChange}
        />
        <Text onPress={this.onToggleSecure} style={s.hideShowText}>
          {this.state.secure ? 'Show' : 'Hide'}
        </Text>
      </View>
    );
  }
}

// STYLES
// =========================
const s = StyleSheet.create({
  hideShowText: {
    fontFamily: fontConfig.semiboldFont, 
    position: 'absolute', 
    right: -18,
    bottom: 7, 
    color: colorConfig.primary, 
    fontSize: 15
  }
});


// PROP TYPES
// =========================
PasswordInput.defaultProps = {
  containerStyle: {
    marginLeft: 0,
    marginRight: 0,
  },
  label: 'Password',
  inputStyle: {
    color: colorConfig.primary,
    height: 38,
    fontSize: 17,
    fontFamily: fontConfig.regularFont,
  },
}


// EXPORT
// =========================
export { PasswordInput };
