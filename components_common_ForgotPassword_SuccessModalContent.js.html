// TOP LEVEL IMPORTS
import React from 'react';
import PropTypes from 'prop-types';
import { View, Text, StyleSheet } from 'react-native';
// COMPONENTS
import RegularButton from '../RegularButton';



// DOCS
// =========================

/**
 * SuccessModalContent, is the content inside of the modal shown after submitting email to "forgot password"
 *
 * @namespace SuccessModalContent
 * @memberof ForgotPasswordForm
 * @prop toggleForm {function} - a function to run after user clicks the "okay" button in the modal, will presumably bring the user back to the login screen after submitting their email
 */



// EXPORTED COMPONENT
// =========================
const SuccessModalContent = ({ toggleForm }) => {
  return (
    <View style={s.container}>
      <View>
        <Text style={s.text}>
          We've sent you a password reset email.
        </Text>
        <RegularButton 
            title="Okay" 
            onPress={() => toggleForm('login')} 
        />
      </View>
    </View>
  );
};



// DOCS
// =========================
const s = StyleSheet.create({
    container: {
        padding: 5,
        height: 100,
        display: 'flex',
        justifyContent: 'center',
        alignItems: 'center',
    },
    text: {
        fontSize: 17, 
        textAlign: 'center', 
        marginBottom: 5
    }
});


// DOCS
// =========================
SuccessModalContent.propTypes = {
    toggleForm: PropTypes.func.isRequired
}

SuccessModalContent.defaultProps = {
    toggleForm: () => {}
}



// EXPORT
// =========================
export default SuccessModalContent;
