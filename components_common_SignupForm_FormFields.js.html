const FormFields = ({ email, password, onChange, form, rules }) => {
  const { getFieldError, getFieldDecorator } = form;
  
  let emailErrors;
  let passwordErrors;
  let confirmPasswordErrors;
  

  const checkPassword = (rule, value, callback) => {
    if (value && value !== form.getFieldValue('password1')) {
      callback('Two passwords that you entered are inconsistent!');
    } else {
      callback();
    }
  };

  const repeatNewPassword = [
    { required: true, message: 'Please repeat the new password!' },
    { validator: checkPassword },
  ];

  return (
    <View>
      {getFieldDecorator('email', {
          validateTrigger: 'onSubmit',
          initialValue: email,
          trigger: 'onChange',
          valuePropName: 'value',
          rules: [...formConfig.defaultEmailRules, ...rules],
        })(
          <EmailInput />
        )}
      <FormInputError 
        fieldName="email" 
        getFieldError={getFieldError} 
      />
      {getFieldDecorator('password1', {
          validateTrigger: 'onSubmit',
          initialValue: password,
          trigger: 'onChange',
          valuePropName: 'value',
          rules: [...formConfig.defaultPasswordRules],
        })(
          <PasswordInput label="Password" />
        )}
      <FormInputError fieldName="password1" getFieldError={getFieldError} />
      
      {getFieldDecorator('password2', {
        validateTrigger: 'onSubmit',
        initialValue: password,
        trigger: 'onChange',
        valuePropName: 'value',
        rules: [...formConfig.defaultPasswordRules, ...repeatNewPassword],
      })(
        <PasswordInput label="Confirm Password" />
      )}
      <FormInputError 
        fieldName="password2" 
        getFieldError={getFieldError} 
      />
    </View>
  );
};

const styles = StyleSheet.create({
