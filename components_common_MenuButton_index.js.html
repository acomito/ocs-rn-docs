// TOP LEVEL IMPORTS
import React from 'react';
import PropTypes from 'prop-types';
import { TouchableOpacity } from 'react-native';
import { MaterialIcons } from '@expo/vector-icons';
import { styleConstants } from '../../../lib/config'


// DESTRUCTURING & CONSTANTS
// ====================================




// DOCS
// =========================
/**
 * @name MenuButton 
 * @description a reusable button component used throughout the application
 * @namespace MenuButton
 * @prop navigation {object} - an object coming from react-navigation
 * @returns {object} - returns a react native component
 */


 // EXPORTED COMPONENT
// ====================================
const MenuButton = ({ navigation }) => {
  return (
    <TouchableOpacity 
      style={{ padding: styleConstants.defaultMargin }} 
      onPress={() => navigation.navigate('DrawerOpen')}
    >
      <MaterialIcons 
        name="menu" 
        size={25} 
        color="#FFF" 
      />
    </TouchableOpacity>
  );
};

// PROPS
// =========================
MenuButton.propTypes = {
  navigation: PropTypes.object.isRequired,
};

MenuButton.defaultProps = {
  navigation: {},
};

// EXPORT
// ====================================
export default MenuButton;
