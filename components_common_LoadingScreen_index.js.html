// TOP LEVEL IMPORTS
import React from 'react';
import PropTypes from 'prop-types';
import { View, Text, ActivityIndicator, StyleSheet } from 'react-native';
//lib
import { colorConfig, stylesConfig, SCREEN_WIDTH } from '../../../lib/config';



// DOCS
// =========================

/**
 * LoadingScreen, is the complete signup form component (made up of several components) 
 * that is used inside of AuthScreen
 *
 * @namespace LoadingScreen
 * @prop loadingMessage {string} - the message shown under the spinner
 * @prop activityIndicatorSize {string} - the size of the ActivityIndicator: https://facebook.github.io/react-native/docs/activityindicator.html#size
 */


// EXPORTED COMPONENT
// =========================
const LoadingScreen = ({ loadingMessage, activityIndicatorSize }) => {
  return (
    <View style={styles.container}>
      <View>
        <ActivityIndicator 
          color={colorConfig.primary} 
          size={activityIndicatorSize} 
        />
        <Text style={styles.header}>
          { loadingMessage }
        </Text>
      </View>
    </View>
  );
};

// STYLES
// =========================
const styles = StyleSheet.create({
  container: {
    flex: 1,
    flexGrow: 1,
    backgroundColor: colorConfig.screenBackground,
    alignItems: 'center',
    justifyContent: 'center',
    width: SCREEN_WIDTH,
  },
  header: {
    fontFamily: stylesConfig.semiboldFont,
    fontSize: 16,
    color: '#888',
    textAlign: 'center',
    marginTop: 10,
  },
});

// PROP
// =========================
LoadingScreen.propTypes = {
  loadingMessage: PropTypes.string,
  activityIndicatorSize: PropTypes.string,
};

LoadingScreen.defaultProps = {
  loadingMessage: 'Loading...',
  activityIndicatorSize: 'large'
};

// EXPORT
// =========================
export default LoadingScreen;
