import React from 'react';
import PropTypes from 'prop-types';
import { View, Dimensions } from 'react-native';
import { colorConfig, stylesConfig, fontConfig } from '../../../lib/config';
import { Button } from 'react-native-elements';


// CONSTANTS & DESTRUCTURING
// =========================
// SHOULD THE BUTTON BE SAME WIDTH AS MARGINS IN LOGIN FORM OR SHOULD IT BE 16dp FROM EACH SIDE?
const SCREEN_WIDTH = Dimensions.get('window').width;
const BUTTON_WIDTH = SCREEN_WIDTH - 104;


// DOC
// =========================
/**
 * RegularButton, a reusable button component used throughout the application
 *
 * @namespace RegularButton
 * @prop buttonStyle {object} - an object that can be passed in to style the button
 * @prop textStyle {object} - an object that can passed in to style the text for the button
 * @prop disabled {boolean} - a true/false value to determine if the button should be disabled or not
 * @prop loading {boolean} - a true/false value to determine if the button should be loading or not
 * @prop title {string} - the title/text to be shown on the button
 * @prop onPress {function} - the function fired onPress
 */


// EXPORTED COMPONENT
// =========================
class RegularButton extends React.PureComponent {
  handleOnPress = () => {
    this.props.onPress();
  };
  render() {

    const { title, disabled, textStyle, buttonStyle, loading } = this.props;
    let defaultTextStyle = { fontSize: 15, fontFamily: fontConfig.semiboldFont };
    let defaultButtonStyle = { width: BUTTON_WIDTH, borderRadius: 8, height: 44 };
    let defaultContainerStyle = { borderRadius: 8, height: 44 };

    return (
      <Button
        onPress={this.handleOnPress}
        title={title}
        raised
        disabled={disabled}
        textStyle={[
          stylesConfig.primaryButtonText,defaultTextStyle, textStyle
        ]}
        buttonStyle={[
          stylesConfig.primaryButton, defaultButtonStyle, buttonStyle
        ]}
        containerViewStyle={defaultContainerStyle}
        loading={loading}
      />
    );

  }
}


// PROP TYPES
// =========================
RegularButton.propTypes = {
  buttonStyle: PropTypes.object,
  textStyle: PropTypes.object,
  disabled: PropTypes.bool,
  loading: PropTypes.bool,
  title: PropTypes.string.isRequired,
  onPress: PropTypes.func,
};

RegularButton.defaultProps = {
  buttonStyle: {},
  textStyle: {},
  disabled: false,
  loading: false,
  title: '',
};


// EXPORT
// =========================
export default RegularButton;
