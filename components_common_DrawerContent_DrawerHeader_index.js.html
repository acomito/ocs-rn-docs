// COMPONENTS
import Avatar from '../../Avatar';

// DOCS
// =========================
/**
 * @namespace DrawerHeader
 * @description component shown inside of the DrawerContent component
 * @memberof DrawerContent
 */



const DrawerHeaderLeft = ({ appStore }) => {
    const { parentsList, selectedParent } = appStore;
    let otherParent = parentsList.filter(item => item.id === selectedParent);
    if (!otherParent[0]) { return null };
    return (
        <View style={s.column}>
            <Avatar 
                size={50}
                backgroundColor='#fff'
                textColor='green'
                onPress={() => {}} 
                parent={{ firstName: 'M', lastName: 'E'}} 
            />
            <Avatar 
                size={50}
                image={otherParent[0].avatar}
                backgroundColor={otherParent[0].color}
                onPress={() => {}} 
                parent={otherParent[0]} 
            />
        </View>
    )
}


const DrawerHeaderRight = ({ appStore, onSelectParent }) => {
    const { parentsList, selectedParent } = appStore;
    let otherParents = parentsList.filter(item => item.id !== selectedParent);
    if (!otherParents) { return null };
    
    return (
        <View style={[s.column, {paddingLeft: 20}]}>
            {otherParents.map((parent, i) => (
                <Avatar 
                    key={parent.id} 
                    backgroundColor={parent.color}
                    onPress={onSelectParent} 
                    parent={parent} 
                />
            ))}
        </View>
    )
}

// EXPORTED COMPONENET
// =========================
class DrawerHeader extends React.PureComponent {
    onSelectParent = (selectedParent) => {
        this.props.onAppStoreChange({ selectedParent });
    }
    render(){
        return (
            <View style={s.container}>
                <DrawerHeaderLeft {...this.props} />
                <DrawerHeaderRight 
                    {...this.props} 
                    onSelectParent={this.onSelectParent} 
                />
            </View>
        )
    }
}
