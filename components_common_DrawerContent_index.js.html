// TOP LEVEL IMPORTS
import React from 'react';
import PropTypes from 'prop-types';
import { SecureStore } from 'expo';
import { View, Text, StyleSheet, ScrollView, TouchableOpacity } from 'react-native';
import { DrawerItems, SafeAreaView } from 'react-navigation';
// LIB
import { colorConfig, stylesConfig } from '../../../lib/config';
// COMPONENTS
import DrawerHeader from './DrawerHeader'


// DOCS
// =========================

/**
 * @namespace DrawerContent
 * @description the drawer used inside authenticated area navigation
 * @prop screenProps {object} - props passed in from react-navigation's top level
 */



// EXPORTED COMPONENT
// ==============================
class DrawerContent extends React.PureComponent {
  renderExtra = () => {
    // Temporary code
    return (
      <View>
        <TouchableOpacity
          style={{ marginTop: 15 }}
          onPress={async () => {
            let result = await SecureStore.deleteItemAsync('token');
            this.props.screenProps.checkToken();
          }}
        >
          <Text>LOGOUT</Text>
        </TouchableOpacity>
        <TouchableOpacity
          style={{ marginTop: 15 }}
          onPress={async () => {
            let result = await SecureStore.deleteItemAsync('onboardingComplete');
            this.props.screenProps.checkOnboarding();
          }}
        >
          <Text>CLEAR ONBOARDING</Text>
        </TouchableOpacity>
      </View>
    )
  }
  render(){

    return (
      <View style={s.outterContainer}>
        <DrawerHeader />
        <ScrollView>
          <SafeAreaView forceInset={{ top: 'always', horizontal: 'never' }}>
            <DrawerItems {...this.props} />
            {this.renderExtra()}
          </SafeAreaView>
        </ScrollView>
      </View>
    );

  }
}


// STYLES
// ==============================
const s = StyleSheet.create({
    outterContainer: {
        flex: 1, 
        paddingTop: 200, 
        position: 'relative'
    }
});


// PROP TYPES
// ==============================

DrawerContent.propTypes = {
  screenProps: PropTypes.object.isRequired
}

DrawerContent.defaultProps = {
  screenProps: {}
}

// EXPORT
// ==============================
export default DrawerContent;
