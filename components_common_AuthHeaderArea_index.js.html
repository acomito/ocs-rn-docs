// TOP LEVEL IMPORTS
import React from 'react';
import PropTypes from 'prop-types';
import { View, Text, StyleSheet, Dimensions } from 'react-native';
// LIB
import { stylesConfig, colorConfig, appConfig, fontConfig, styleConstants } from '../../../lib/config';
// COMPONENTS
import LinkText from '../LinkText';
import { HeaderText } from '../../text';



// DOCS
// =========================
/**
 *
 * @namespace AuthHeaderArea
 * @description component that can be re-used inside of the header of the AuthScreens sub-views (LoginForm, SignupForm, etc.)
 * @prop toggleForm {function} - function to run when text is pressed
 * @prop authLinkLabel {string} - string to be used in the link
 * @prop authHeaderLabel {string} - string to be used in the header/title of the page
 * 
 */


// EXPORTED COMPONENT
// =========================
const AuthHeaderArea = ({ authLinkLabel, toggleForm, authHeaderLabel }) => {
  return (
    <View style={s.container}>
      {authLinkLabel && (
        <View style={s.linkContainer}>
          <LinkText 
            label={authLinkLabel} 
            toggleForm={toggleForm} 
          />
        </View>
      )}
      <HeaderText>
        { authHeaderLabel }
      </HeaderText>
    </View>
  )
}


// STYLES
// =========================
const s = StyleSheet.create({
  linkContainer: {
    position: 'absolute', 
    right: -18, 
    top: 15
  },
  container: {
      maxHeight: styleConstants.topAreaHeight, 
      height: styleConstants.topAreaHeight,
      display: 'flex',
      justifyContent: 'flex-end',
      alignItems: 'flex-start',
      width: Dimensions.get('window').width-96, 
      position: 'relative'
  }
});


// PROP TYPES
// =========================
AuthHeaderArea.propTypes = {
  authLinkLabel: PropTypes.oneOfType([
    PropTypes.string,
    PropTypes.object
  ]),
  toggleForm: PropTypes.func,
  authHeaderLabel: PropTypes.string.isRequired, 
}

AuthHeaderArea.defaultProps = {
  authLinkLabel: null, 
  toggleForm: () => {}, 
  authHeaderLabel: ''
}


// EXPORT
// =========================
export default AuthHeaderArea;
