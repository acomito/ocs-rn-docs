// TOP LEVEL IMPORTS
import React from 'react';
import PropTypes from 'prop-types';
import { View, StyleSheet } from 'react-native';
// LIB
import { stylesConfig, colorConfig, appConfig, fontConfig, formConfig } from '../../../lib/config';
// COMPONENTS
import { PasswordInput, EmailInput } from '../../form-inputs';
//
import FormInputError from '../FormInputError';


// DOCS
// =========================
/**
 * SignupFormFields, is a set of form fields used inside of SignupForm
 *
 * @namespace SignupFormFields
 * @memberof SignupForm
 * @prop form {object} - the default form object from rc-form
 * @prop email {string} - the default email string
 * @prop password {string} - the default password string
 * @prop onChange {function} - provided by rc-form, will update rc-form state
 */



// EXPORTED COMPONENT
// =========================
const SignupFormFields = ({ email, password, onChange, form, rules }) => {
  const { getFieldError, getFieldDecorator } = form;
  
  let emailErrors;
  let passwordErrors;
  let confirmPasswordErrors;
  

  const checkPassword = (rule, value, callback) => {
    if (value && value !== form.getFieldValue('password1')) {
      callback('Two passwords that you entered are inconsistent!');
    } else {
      callback();
    }
  };

  const repeatNewPassword = [
    { required: true, message: 'Please repeat the new password!' },
    { validator: checkPassword },
  ];

  return (
    <View>
      {getFieldDecorator('email', {
          validateTrigger: 'onSubmit',
          initialValue: email,
          trigger: 'onChange',
          valuePropName: 'value',
          rules: [...formConfig.defaultEmailRules, ...rules],
        })(
          <EmailInput />
        )}
      <FormInputError 
        fieldName="email" 
        getFieldError={getFieldError} 
      />
      {getFieldDecorator('password1', {
          validateTrigger: 'onSubmit',
          initialValue: password,
          trigger: 'onChange',
          valuePropName: 'value',
          rules: [...formConfig.defaultPasswordRules],
        })(
          <PasswordInput label="Password" />
        )}
      <FormInputError 
        fieldName="password1"
        getFieldError={getFieldError} 
      />
      {getFieldDecorator('password2', {
        validateTrigger: 'onSubmit',
        initialValue: password,
        trigger: 'onChange',
        valuePropName: 'value',
        rules: [...formConfig.defaultPasswordRules, ...repeatNewPassword],
      })(
        <PasswordInput label="Confirm Password" />
      )}
      <FormInputError 
        fieldName="password2" 
        getFieldError={getFieldError} 
      />
    </View>
  );
};




// PROP TYPES
// =========================
SignupFormFields.propTypes = {
  form: PropTypes.object, 
  email: PropTypes.string,
  password: PropTypes.string,
  onChange: PropTypes.func,
  rules: PropTypes.array,
}


SignupFormFields.defaultProps = {
  form: {},
  email: '', 
  password: '', 
  onChange: () => {}, 
  rules: []

}


// EXPORT
// =========================
export default SignupFormFields;
