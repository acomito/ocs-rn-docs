import React from 'react';
import PropTypes from 'prop-types';
import { Text, StyleSheet } from 'react-native';
import { colorConfig, fontConfig } from '../../lib/config'


// DOCS
// =========================

/**
 * @namespace HeaderText
 * @description reusable header text typically used as a sort of h1 page title
 * @prop children {object} 
 * @prop textStyle {object}
 * 
 */


// EXPORTED COMPONENT
// =========================
const HeaderText = ({ children, textStyle }) => {

  if (!children) { return null };
 
  return (
    <Text style={[ s.defaultTextStyles, textStyle ]}>
      { children }
    </Text>
  );

};


// STYLES
// =========================
const s = StyleSheet.create({
  defaultTextStyles: { 
    fontFamily: fontConfig.boldFont, 
    color: colorConfig.black1, 
    fontSize: 24,
    margin: 0,
    textAlign: 'left' 
  }
});



// PROP TYPES
// =========================

HeaderText.propTypes = {
  textStyle: PropTypes.object,
  children: PropTypes.node.isRequired,
}

HeaderText.defaultProps = {
  textStyle: {},
  children: null
}


// EXPORT
// =========================
export { HeaderText } ;
