// TOP LEVEL IMPORTS
import React from 'react';
import PropTypes from 'prop-types';
import { View, TextInput, TouchableWithoutFeedback, Keyboard } from 'react-native';
// COMPONENTS
import FormLabel from './FormLabel';




// DOCS
// =========================
/**
 *
 * @namespace DescriptionInput
 * @description reusable textarea input
 * @prop onChange {function} 
 * @prop value {string}
 * @prop placeholder {string}
 * @prop label {string}
 * 
 */


// EXPORTED COMPONENT
// ===================================
class DescriptionInput extends React.Component {
  render() {
    const { onChange, value, placeholder, label } = this.props;

    return (
      <TouchableWithoutFeedback onPress={Keyboard.dismiss}>
        <View>
          {label && <FormLabel label={label} />}
          <TextInput
            clear
            placeholder={placeholder}
            rows={4}
            value={value}
            onChange={val => onChange(val)}
          />
        </View>
      </TouchableWithoutFeedback>
    );
  }
}

// PROPS
// ===================================
DescriptionInput.propTypes = {
  label: PropTypes.string,
  form: PropTypes.object,
  message: PropTypes.string,
  placeholder: PropTypes.string,
  required: PropTypes.bool,
};

DescriptionInput.defaultProps = {
  label: 'Description',
  placeholder: 'Type your message here...',
  message: 'Please choose a status!',
  required: true,
};

// EXPORT
// ===================================
export { DescriptionInput };
