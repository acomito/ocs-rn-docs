// TOP LEVEL IMPORTS
import React from 'react';
import PropTypes from 'prop-types';
import { View, Text, StyleSheet } from 'react-native';
// lib
import { colorConfig, fontConfig } from '../../lib/config';



// DOCS
// =========================
/**
 *
 * @namespace FormLabel
 * @description reusable form label for inputs
 * @prop label {string}
 * 
 */


// EXPORTED COMPONENT
// ===================================
const FormLabel = ({ label }) => {
  return (
    <View style={s.labelContainer}>
      <Text style={s.labelStyle}>
        { label }
      </Text>
    </View>
  );
};

// STYLES
// ===================================
const s = StyleSheet.create({
  labelContainer: {
    padding: 4,
    marginTop: 5,
  },
  labelStyle: {
    color: colorConfig.primaryDark,
    fontFamily: fontConfig.boldFont,
    fontWeight: '500',
  },
});

// PROPS
// ===================================
FormLabel.propTypes = {
  label: PropTypes.string,
};

FormLabel.defaultProps = {
  label: '',
};

// EXPORT
// ===================================
export default FormLabel;
