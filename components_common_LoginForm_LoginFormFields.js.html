// TOP LEVEL IMPORTS
import React from 'react';
import PropTypes from 'prop-types';
import { View } from 'react-native';
// LIB
import { 
  stylesConfig, 
  fontConfig, 
  colorConfig, 
  appConfig, 
  formConfig
} from '../../../lib/config';
import { PasswordInput, EmailInput } from '../../form-inputs';
// COMPONENTS
import FormInputError from '../FormInputError';



/**
 * LoginFormFields, is a set of form fields used inside of SignupForm
 *
 * @namespace LoginFormFields
 * @memberof LoginForm
 * @prop form {object} - the default form object from rc-form
 * @prop email {string} - the default email string
 */

const LoginFormFields = ({ email, password, onChange, form, rules = [] }) => {

  const { getFieldError, getFieldDecorator } = form;

  let emailErrors;
  let passwordErrors;
  

  return (
    <View>
      {getFieldDecorator('email', {
        validateTrigger: 'onSubmit',
        initialValue: email,
        trigger: 'onChange',
        valuePropName: 'value',
        rules: [...formConfig.defaultEmailRules, ...rules],
      })(
        <EmailInput />
      )}
      <FormInputError 
        fieldName="email" 
        getFieldError={getFieldError} 
      />
      {getFieldDecorator('password', {
        validateTrigger: 'onSubmit',
        initialValue: password,
        trigger: 'onChange',
        valuePropName: 'value',
        rules: [...formConfig.defaultPasswordRules],
      })(
        <PasswordInput />
      )}
      <FormInputError 
        fieldName="password" 
        getFieldError={getFieldError} 
      />
    </View>
  );
};




// PROP-TYPES
// ========================================
LoginFormFields.propTypes = {
  form: PropTypes.object, 
  email: PropTypes.string,
  password: PropTypes.string,
  onChange: PropTypes.func,
  rules: PropTypes.array,
}

LoginFormFields.defaultProps = {
  rules: [],
  form: {},
  onChange: () => {},
  password: '',
  email: ''
}


// FINAL EXPORT
// ========================================
export default LoginFormFields;
